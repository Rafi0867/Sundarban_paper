// water-blue; forest-deep green; builduo-red; aquaculture-orange; low density vegetation/baren land/Others-pink;
// forest means deep vegetation
// others include everything except water, forest, aquaculture and  build ups
//..................................................................................................................




// preamble
var L7 = ee.ImageCollection("LANDSAT/LE07/C02/T1_TOA")
  .filterBounds(ROI)
  .filterDate("2000-01-01", "2000-12-31")
  .filterMetadata('CLOUD_COVER','less_than', 1)
  .mean()
  .clip(ROI);


// identifying different variables that helps to create the classified image
Map.addLayer(L7, {bands:["B5" , "B4" , "B3"]});
  var training_points     = water.merge(forest).merge(builtup).merge(aquaculture).merge(others);
  var training_data       = L7.sampleRegions({collection:training_points,properties:['LC'],scale:30});
  var classifier          = ee.Classifier.smileCart();
  var classifier          = classifier.train({features:training_data,classProperty: 'LC',inputProperties:["B1", "B2", "B3", "B4", "B5", "B7"]});
  var classified_image    = L7.classify(classifier);
  var trainingData        = training_data.randomColumn();
  var trainSet            = trainingData.filter(ee.Filter.lessThan('random', 0.8));
  var testSet             = trainingData.filter(ee.Filter.greaterThanOrEquals('random', 0.8));
  
// specify attibutes of classified image
Map.addLayer(classified_image,{min:0, max:3, palette:[ 'blue', 'green', 'red', 'orange','pink']},'classified image');

// specifying attributes of confusion matrix
var confusionMatrix = ee.ConfusionMatrix(testSet.classify(classifier)
.errorMatrix
  (
    {
      actual:     'LC',
      predicted:  'classification'
    }
  )
);

// print some model diagonestics
print('Confusion Matrix', confusionMatrix);
print("Overall Accuracy:", confusionMatrix.accuracy());

// Export image as .tiff file
Export.image.toDrive(
  {
    image:        classified_image,
    description:  'ExportedData',
    folder:       "asset",
    region:       ROI,
    scale:        30,
    maxPixels:    1e13,
  }
);
