//  Random Forest Supervised Classification Algorithm for sundarban


// STEP 1
// load landsat 9 surface reflectance data
var dataset = ee.ImageCollection("LANDSAT/LE07/C02/T1_TOA")
    .filterDate('2005-01-01', '2005-12-31')
    .filterBounds(ROI)
    .filterMetadata('CLOUD_COVER', 'less_than', 5)
    ;
    
    
// //  Applies the scalling factors
// function applyScaleFactors(image){
//   var opticalBands = image.select('SR_B.').multiply(0.0000275).add(-0.2);
//   var thermalBands = image.select('SR_B.').multiply(0.00341802).add(149.0);
//   return image.addBands(opticalBands, null, true)
//               .addBands(opticalBands, null, true);
// }


// //  Rescaling the image with median value
// var rescale = dataset.map(applyScaleFactors);
var image = dataset.mean();

// creating image visualization parameter
var visualization = {
  bands: ["B5" , "B4" , "B3"],
  min: 0.0,
  max: 0.3,
};

// displaying image
Map.addLayer(image, visualization, 'Landsat 8');
Map.centerObject(ROI);



// STEP 2
// create training data
var training = others.merge(forest).merge(water).merge(builtup).merge(aquaculture);
print(training);

// Overlay trainign data into the Landsat data
var label = 'Class';
var bands = ["B1", "B2", "B3", "B4", "B5", "B7"];
var input = image.select(bands);

//  Overlay the training data to imegery to get training data
var trainImage = input.sampleRegions({
    collection: training,
    properties: [label],
    scale: 30
})

//print training image
print(trainImage);


// STEP 3
// Classification model
//  declering variables for running random forest model
var trainingData = trainImage.randomColumn();
var trainSet = trainingData.filter(ee.Filter.lessThan('random', 0.8));
var testSet = trainingData.filter(ee.Filter.greaterThanOrEquals('random', 0.8));


//  Make a Random Forest Model classifier and train it
var classifier = ee.Classifier.smileRandomForest(10)
    .train({
      features: trainSet,
      classProperty: label,
      inputProperties: bands
    });
    
    
// classify the image
var Classified = input.classify(classifier);


// define a color pallet for the classification model
var lcpallet = [
    'blue', //water
    'red', //urban
    'green', //forest
    'orange', //crop
    'pink', //barren
];

// add classified layer to the map
Map.addLayer(Classified.clip(ROI), 
  {
    pallet: lcpallet, 
    min:0, 
    max:4
  }, 
  'classification'
);



// STEP 4
// classify the tesing set to get the confusion matrix
var confusionmatrix = ee.ConfusionMatrix(testSet.classify(classifier)
  .errorMatrix({
    actual: 'Class',
    predicted: 'classification'
  }));
  
//  Print results of matrix
print('Confusion_matrix:', confusionmatrix);
print('Accuracy:', confusionmatrix.accuracy());


// STEP 5
// Exprot classified image
Export.image.toDrive({
  image: Classified,
  description: 'Sundarban_rf_05',
  region: ROI,
  scale: 30,
  maxPixels: 1e13,
});
